#Computer Trivia Game#These lines welcome the user to the trivia game and explain its purpose.print("Welcome to my computer trivia game!")print("This test your knowledge of computer!")#This line prompts the user to indicate if they want to play the game by typing their response.question = input("Do you want to play? ")#This checks if the user's input (converted to lowercase) is not "yes." If the answer is anything else, it informs the user that they can only answer "yes" or "no" and then exits the game.if question.lower() != "yes":    print("You can only answer yes or no")    quit()#This line asks the user for their age and converts the input from a string to an integer.age= int(input("How old are you? "))#This checks if the user's age is 17 or younger. If true, it informs the user that they are not qualified to play and exits the game.#If the user is older than 17, this line acknowledges their eligibility to play.if age <= 17:    print ("You're not qualified")    quit ()else:    print("Okay! Let's play :)")#Initializes a variable score to keep track of the number of correct answers.score = 0#This block asks the user what "CPU" stands for. It checks if the answer (in lowercase) matches "central processing unit." If correct, it increments the score by 1; otherwise, it informs the user that their answer is incorrect.#The same pattern is repeated for questions about "GPU," "RAM," and "PSU," each checking for the correct answer and updating the score accordingly.answer = input("What does CPU stand for? ")if answer.lower() == "central processing unit":    print('Correct!')    score += 1else:    print("Incorrect!")answer = input("What does GPU stand for? ")if answer.lower() == "graphics processing unit":    print('Correct!')    score += 1else:    print("Incorrect!")answer = input("What does RAM stand for? ")if answer.lower() == "random access memory":    print('Correct!')    score += 1else:    print("Incorrect!")answer = input("What does PSU stand for? ")if answer.lower() == "power supply":    print('Correct!')    score += 1else:    print("Incorrect!")#After all questions, these lines display the total number of correct answers and the percentage score based on the total number of questions (4)print("You got " + str(score) + " questions correct!")print("You got " + str((score / 4) * 100) + "%.")print ("Try again !")
